\documentclass{article}
\usepackage{multicol}
\usepackage[utf8]{inputenc}

\makeatletter
{\catcode`\^^M=\active %
 \gdef\obeylinesandstanzas{%
  \catcode`\^^M=\active %
  \def^^M{\futurelet\next\check@newline}%
 }%
 \gdef\check@newline{%
  \ifx\next^^M%
    \expandafter\stanza@break%
  \else%
    \\%
  \fi%
 }%
 \gdef\stanza@break^^M{%
  \par\end{minipage}\vspace{1em}\par\noindent\textbf{[BREAK]}\begin{minipage}{\columnwidth}\raggedright%
 }%
}
\makeatother

\newcommand{\inputpoem}[1]{%
  \begingroup
  \obeylinesandstanzas
  \noindent\begin{minipage}{\columnwidth}\raggedright
  \input{#1}\par\end{minipage}%
  \endgroup
}

\begin{document}
\begin{multicols}{3}
\inputpoem{testpoem.tex}
\end{multicols}
\end{document}
