\documentclass{article}
\usepackage{multicol}
\usepackage[utf8]{inputenc}

\makeatletter
{\catcode`\^^M=\active %
 \gdef\obeylinesandstanzas{%
  \catcode`\^^M=\active %
  \def^^M{\futurelet\next\check@newline}%
 }%
 \gdef\check@newline{%
  \ifx\next^^M%
    \expandafter\stanza@break%
  \else%
    \\%
  \fi%
 }%
 \gdef\stanza@break^^M{%
  \par\vspace{1em}\noindent%
 }%
}
\makeatother

\newcommand{\inputpoem}[1]{%
  \begingroup
  \interlinepenalty=10000
  \obeylinesandstanzas
  \noindent\input{#1}%
  \endgroup
}

\begin{document}
\begin{multicols}{3}
\inputpoem{testpoem.tex}
\end{multicols}
\end{document}
